# Clean column names for easier access
# First, remove ' ($)' or ' ($)' suffix using regex
df.columns = df.columns.str.replace(r'\s*\(\$?\)$', '', regex=True)
# Then replace spaces with underscores and convert to uppercase
df.columns = df.columns.str.replace(' ', '_').str.upper()

# Correcting a potential typo if 'STOCKGTP' was intended to be 'STOCKGPT'
df = df.rename(columns={'STOCKGTP_CURRENT_PRICE': 'STOCKGPT_CURRENT_PRICE'})

import numpy as np

def calculate_mae(actual, predicted):
    return np.mean(np.abs(actual - predicted))

def calculate_rmse(actual, predicted):
    return np.sqrt(np.mean((actual - predicted)**2))

def calculate_mape(actual, predicted):
    # Avoid division by zero by filtering out actual values that are 0
    non_zero_actual = actual[actual != 0]
    non_zero_predicted = predicted[actual != 0]

    if len(non_zero_actual) == 0:
        return np.nan # Cannot calculate MAPE if all actual values are zero
    return np.mean(np.abs((non_zero_actual - non_zero_predicted) / non_zero_actual)) * 100


# Prepare to store results
analysis_results = []

# Iterate through each unique stock ticker
for stock_ticker in df['STOCK'].unique():
    stock_df = df[df['STOCK'] == stock_ticker].copy()

    # Ensure the columns are numeric. Coerce errors will turn non-numeric into NaN.
    stock_df['YAHOO_FINANCE_CURRENT_PRICE'] = pd.to_numeric(stock_df['YAHOO_FINANCE_CURRENT_PRICE'], errors='coerce')
    stock_df['STOCKGPT_WEIGHTED_FAIR_VALUE_(PREDICTED)'] = pd.to_numeric(stock_df['STOCKGPT_WEIGHTED_FAIR_VALUE_(PREDICTED)'], errors='coerce')

    # Drop rows where either actual or predicted is NaN after conversion
    stock_df.dropna(subset=['YAHOO_FINANCE_CURRENT_PRICE', 'STOCKGPT_WEIGHTED_FAIR_VALUE_(PREDICTED)'], inplace=True)

    if not stock_df.empty:
        actual_values = stock_df['YAHOO_FINANCE_CURRENT_PRICE']
        predicted_values = stock_df['STOCKGPT_WEIGHTED_FAIR_VALUE_(PREDICTED)']

        mae = calculate_mae(actual_values, predicted_values)
        rmse = calculate_rmse(actual_values, predicted_values)
        mape = calculate_mape(actual_values, predicted_values)

        # Get additional data for the table
        last_timestamp = stock_df['TIMESTAMP'].iloc[-1] if not stock_df.empty else None
        mean_weighted_fair_value = stock_df['STOCKGPT_WEIGHTED_FAIR_VALUE_(PREDICTED)'].mean()
        mean_yahoo_current_price = stock_df['YAHOO_FINANCE_CURRENT_PRICE'].mean()

        analysis_results.append({
            'Stock': stock_ticker,
            'MAE': mae,
            'RMSE': rmse,
            'MAPE': mape,
            'Last_Timestamp': last_timestamp,
            'Mean_StockGPT_Fair_Value': mean_weighted_fair_value,
            'Mean_Yahoo_Current_Price': mean_yahoo_current_price
        })

# Convert results to a DataFrame for tabular display
results_df = pd.DataFrame(analysis_results)

# Reorder columns as requested by the user
results_df = results_df[['Stock', 'Last_Timestamp', 'Mean_StockGPT_Fair_Value', 'Mean_Yahoo_Current_Price', 'MAE', 'RMSE', 'MAPE']]

# Display the results with centered alignment
display(results_df.style.set_properties(**{'text-align': 'center'}))
